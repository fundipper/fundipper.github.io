package main

import (
	"fmt"
	"log"
	"os"
	"time"

	"github.com/goburrow/modbus"
)

func main() {
	handler := modbus.NewRTUClientHandler("/dev/ttyAMA0")
	handler.BaudRate = 9600
	handler.DataBits = 8
	handler.Parity = "N"
	handler.Timeout = 512 * time.Millisecond
	handler.Logger = log.New(os.Stdout, "test: ", log.LstdFlags)
	handler.SlaveId = 4
	err := handler.Connect()
	if err != nil {
		fmt.Println(err)
	}
	defer handler.Close()

	client := modbus.NewClient(handler)
	result, err := client.ReadHoldingRegisters(73, 1)
	if err != nil {
		fmt.Println(err)
	}
	fmt.Println(result)
}
